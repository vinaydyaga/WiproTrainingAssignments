DROP TABLE IF EXISTS Employees;
PRAGMA foreign_keys =ON;
CREATE TABLE Employees(
    employee_id INTEGER,
    first_name TEXT,
    last_name TEXT,
    department_id INTEGER,
    PRIMARY KEY (employee_id),
    FOREIGN KEY (department_id)
    REFERENCES Departments(department_id)
);
