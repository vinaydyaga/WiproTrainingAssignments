SELECT store_id, film_id, title, rental_count
FROM (SELECT i.store_id,f.film_id,f.title, COUNT(r.rental_id)AS rental_count,
ROW_NUMBER()OVER(PARTITION BY i.store_id ORDER  BY COUNT(r.rental_id)DESC)AS rn
FROM rental r
JOIN inventory i
ON r.inventory_id=i.inventory_id
JOIN film f
ON i.film_id=f.film_id
GROUP BY i.store_id, f.film_id,f.title)ranked
WHERE rn<=3
ORDER BY store_id, rental_count DESC;
